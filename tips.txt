# 현재 환경에서 설치된 패키지 출력
pip freeze

# 파일 생성 명령어
cmd - type nul > [파일명]
bash - touch [파일명]

# bash 가상환경 진입
source venv/Scripts/activate

# 환경변수 설정 + FLASK_DEBUG=1로 설정하면 코드 바로 반영
- cmd
set FLASK_ENV=development
set FLASK_APP=hello.py
set FLASK_DEBUG=1
- bash
export [환경변수명]=[값]
export FLASK_ENV=development
export FLASK_APP=hello.py
export FLASK_DEBUG=1
